        "# @title CardTracker: 相手手札の推論エンジン\n",
        "class CardTracker:\n",
        "    \"\"\"パス履歴から相手の手札可能性を推論\"\"\"\n",
        "\n",
        "    def __init__(self, state, my_player_num):\n",
        "        self.players_num = state.players_num\n",
        "        self.my_player_num = my_player_num\n",
        "        self.all_cards = [Card(s, n) for s in Suit for n in Number]\n",
        "        \n",
        "        # possible[p] = プレイヤーpが持ちうるカード集合\n",
        "        self.possible = [set(self.all_cards) for _ in range(self.players_num)]\n",
        "        self.pass_counts = [0] * self.players_num\n",
        "        \n",
        "        # 場に出たカードは誰も持たない\n",
        "        self._apply_field(state)\n",
        "        \n",
        "        # 自分の手札は確定\n",
        "        my_hand = set(state.players_cards[my_player_num])\n",
        "        for p in range(self.players_num):\n",
        "            if p == my_player_num:\n",
        "                self.possible[p].intersection_update(my_hand)\n",
        "            else:\n",
        "                self.possible[p].difference_update(my_hand)\n",
        "        \n",
        "        self.out_player = set(state.out_player)\n",
        "\n",
        "    def _apply_field(self, state):\n",
        "        \"\"\"場に出たカードを除外\"\"\"\n",
        "        for s_idx, s in enumerate(Suit):\n",
        "            for n_idx, n in enumerate(Number):\n",
        "                if state.field_cards[s_idx][n_idx] == 1:\n",
        "                    card = Card(s, n)\n",
        "                    for p in range(self.players_num):\n",
        "                        self.possible[p].discard(card)\n",
        "\n",
        "    def observe_action(self, state, player, action, is_pass):\n",
        "        \"\"\"行動観測で可能性を更新\"\"\"\n",
        "        if player in self.out_player:\n",
        "            return\n",
        "\n",
        "        if is_pass:\n",
        "            self.pass_counts[player] += 1\n",
        "            # パス時、出せるカードを持っていないと推論\n",
        "            legal = state.legal_actions()\n",
        "            for c in legal:\n",
        "                self.possible[player].discard(c)\n",
        "        elif action is not None:\n",
        "            # カードを出したら全員が持っていない\n",
        "            for p in range(self.players_num):\n",
        "                self.possible[p].discard(action)\n",
        "\n",
        "    def mark_out(self, player):\n",
        "        self.out_player.add(player)\n",
        "        self.possible[player].clear()\n",
        "\n",
        "    def get_player_weight(self, player):\n",
        "        \"\"\"パス回数に基づく重み（0.5～1.0）\"\"\"\n",
        "        weight = 1.0 - (self.pass_counts[player] / 4.0) * 0.5\n",
        "        return max(0.5, weight)"

        "            start_flags[p] = self.choice_seven(hand=self.players_cards[p], player=p)\n",

        "# @title HybridStrongestAI: PIMC法を用いた戦略的AI\n",
        "class HybridStrongestAI:\n",
        "    \"\"\"PIMC法を用いた戦略的AI\n",
        "    \n",
        "    Phase 1: 推論エンジン（CardTracker）で相手手札を推測\n",
        "    Phase 2: 確定化（Determinization）で複数の仮想世界を生成\n",
        "    Phase 3: 各仮想世界でプレイアウトして勝率を評価\n",
        "    \n",
        "    改善点:\n",
        "    - PASS候補の完全除外\n",
        "    - 重み付け確定化\n",
        "    - 適応的ロールアウトポリシー\n",
        "    \"\"\"\n",
        "    \n",
        "    # 確定化リトライの上限回数\n",
        "    DETERMINIZE_RETRY = 30\n",
        "    def __init__(self, my_player_num, simulation_count=300):\n",
        "        self.my_player_num = my_player_num\n",
        "        self.simulation_count = simulation_count\n",
        "        self._in_simulation = False\n",
        "\n",
        "    def get_action(self, state):\n",
        "        \"\"\"行動を決定\"\"\"\n",
        "        # シミュレーション中は軽量なロールアウトポリシー\n",
        "        if self._in_simulation:\n",
        "            return self._rollout_policy_action(state)\n",
        "\n",
        "        my_actions = state.my_actions()\n",
        "        if not my_actions:\n",
        "            return None, 1\n",
        "\n",
        "        # Phase 1改善: PASS候補を除外（合法手があれば必ず打つ）\n",
        "        candidates = list(my_actions)\n",
        "\n",
        "        if len(candidates) == 1:\n",
        "            return candidates[0], 0\n",
        "\n",
        "        # Phase 1: 履歴から相手手札を推論\n",
        "        tracker = self._build_tracker_from_history(state)\n",
        "        \n",
        "        # Phase 2 & 3: 各候補をシミュレーションで評価\n",
        "        action_scores = {action: 0 for action in candidates}\n",
        "\n",
        "        for _ in range(self.simulation_count):\n",
        "            # Phase 2: 確定化（推論に基づき相手手札を生成）\n",
        "            determinized_state = self._create_determinized_state_with_constraints(state, tracker)\n",
        "\n",
        "            for first_action in candidates:\n",
        "                sim_state = determinized_state.clone()\n",
        "                sim_state.next(first_action, 0)\n",
        "\n",
        "                # Phase 3: プレイアウト\n",
        "                winner = self._playout(sim_state)\n",
        "\n",
        "                if winner == self.my_player_num:\n",
        "                    action_scores[first_action] += 1\n",
        "                elif winner != -1:\n",
        "                    action_scores[first_action] -= 1\n",
        "\n",
        "        best_action = max(action_scores, key=action_scores.get)\n",
        "        return best_action, 0\n",
        "\n",
        "    def _rollout_policy_action(self, state):\n",
        "        \"\"\"軽量なロールアウトポリシー\n",
        "        \n",
        "        1. 端優先（A/K）：トンネルを活用\n",
        "        2. Safe優先：連続して出せる札を優先\n",
        "        3. ランダム選択\n",
        "        \"\"\"\n",
        "        my_actions = state.my_actions()\n",
        "        if not my_actions:\n",
        "            return None, 1\n",
        "\n",
        "        # 1. 端優先\n",
        "        ends = [a for a in my_actions if a.number in (Number.ACE, Number.KING)]\n",
        "        if ends:\n",
        "            return random.choice(ends), 0\n",
        "\n",
        "        # 2. Safe優先\n",
        "        hand_strs = [str(c) for c in state.players_cards[state.turn_player]]\n",
        "        safe = [a for a in my_actions if self._is_safe_move(a, hand_strs)]\n",
        "        if safe:\n",
        "            return random.choice(safe), 0\n",
        "\n",
        "        # 3. ランダム\n",
        "        return random.choice(my_actions), 0\n",
        "\n",
        "    def _build_tracker_from_history(self, state):\n",
        "        \"\"\"履歴を再生して推論\"\"\"\n",
        "        tracker = CardTracker(state, self.my_player_num)\n",
        "\n",
        "        # 盤面のみ再現する軽量state\n",
        "        replay_state = State(\n",
        "            players_num=state.players_num,\n",
        "            field_cards=np.zeros((4, 13), dtype='int64'),\n",
        "            players_cards=[Hand([]) for _ in range(state.players_num)],\n",
        "            turn_player=0,\n",
        "            pass_count=[0] * state.players_num,\n",
        "            out_player=[],\n",
        "            history=[],\n",
        "        )\n",
        "\n",
        "        # 履歴から開始プレイヤーを復元\n",
        "        start_player = None\n",
        "        for (p0, a0, pf0) in state.history:\n",
        "            if pf0 == 0 and isinstance(a0, Card) and a0 == Card(Suit.DIAMOND, Number.SEVEN):\n",
        "                start_player = p0\n",
        "                break\n",
        "        replay_state.turn_player = start_player if start_player is not None else 0\n",
        "\n",
        "        # 履歴を再生\n",
        "        for (p, a, pf) in state.history:\n",
        "            # 観測\n",
        "            tracker.observe_action(replay_state, p, a, is_pass=(pf == 1 or a is None))\n",
        "\n",
        "            # 盤面更新\n",
        "            if pf == 1 or a is None:\n",
        "                replay_state.pass_count[p] += 1\n",
        "                if replay_state.pass_count[p] > 3 and p not in replay_state.out_player:\n",
        "                    replay_state.out_player.append(p)\n",
        "                    tracker.mark_out(p)\n",
        "            else:\n",
        "                if a is not None:\n",
        "                    try:\n",
        "                        replay_state.put_card(a)\n",
        "                    except:\n",
        "                        pass\n",
        "\n",
        "            # 次手番へ\n",
        "            original = replay_state.turn_player\n",
        "            for i in range(1, replay_state.players_num + 1):\n",
        "                np_ = (original + i) % replay_state.players_num\n",
        "                if np_ not in replay_state.out_player:\n",
        "                    replay_state.turn_player = np_\n",
        "                    break\n",
        "\n",
        "        return tracker\n",
        "\n",
        "    def _create_determinized_state_with_constraints(self, original_state, tracker):\n",
        "        \"\"\"重み付け確定化: 推論制約を満たすように相手手札を生成\"\"\"\n",
        "        base = original_state.clone()\n",
        "\n",
        "        # 相手のカードプール\n",
        "        pool = []\n",
        "        for p in range(base.players_num):\n",
        "            if p != self.my_player_num:\n",
        "                pool.extend(base.players_cards[p])\n",
        "\n",
        "        # いったん空にして再配分\n",
        "        for p in range(base.players_num):\n",
        "            if p != self.my_player_num:\n",
        "                base.players_cards[p] = Hand([])\n",
        "\n",
        "        need = {p: len(original_state.players_cards[p]) for p in range(base.players_num) if p != self.my_player_num}\n",
        "\n",
        "        # 確定化を複数回リトライ（値は DETERMINIZE_RETRY で管理）\n",
        "        # 制約を満たす割り当てを試行（過去の検証でこの回数で十分収束）\n",
        "        for _ in range(self.DETERMINIZE_RETRY):\n",
        "            random.shuffle(pool)\n",
        "            remain = list(pool)\n",
        "            hands = {p: [] for p in need.keys()}\n",
        "            ok = True\n",
        "\n",
        "            # 各プレイヤーに可能なカードを割り当て\n",
        "            for p in need.keys():\n",
        "                k = need[p]\n",
        "                if k == 0:\n",
        "                    continue\n",
        "\n",
        "                possible_list = [c for c in remain if c in tracker.possible[p]]\n",
        "                \n",
        "                if len(possible_list) < k:\n",
        "                    ok = False\n",
        "                    break\n",
        "                \n",
        "                chosen = possible_list[:k]\n",
        "                hands[p].extend(chosen)\n",
        "                chosen_set = set(chosen)\n",
        "                remain = [c for c in remain if c not in chosen_set]\n",
        "\n",
        "            if not ok:\n",
        "                continue\n",
        "\n",
        "            # 残りを分配\n",
        "            if remain:\n",
        "                ps = list(need.keys())\n",
        "                idx = 0\n",
        "                for c in remain:\n",
        "                    hands[ps[idx % len(ps)]].append(c)\n",
        "                    idx += 1\n",
        "\n",
        "            # 反映\n",
        "            for p, cards in hands.items():\n",
        "                base.players_cards[p] = Hand(cards)\n",
        "\n",
        "            return base\n",
        "\n",
        "        # フォールバック: ランダム確定化\n",
        "        return self._create_determinized_state(original_state, pool)\n",
        "\n",
        "    def _is_safe_move(self, card, hand_card_strs):\n",
        "        \"\"\"次のカードを自分が持っていればSafe\"\"\"\n",
        "        val = card.number.val\n",
        "        suit = card.suit\n",
        "        \n",
        "        if val == 1 or val == 13:\n",
        "            return True\n",
        "            \n",
        "        next_target_val = -1\n",
        "        if val < 7:\n",
        "            next_target_val = val - 1\n",
        "        elif val > 7:\n",
        "            next_target_val = val + 1\n",
        "        else:\n",
        "            return False\n",
        "        \n",
        "        target_number = None\n",
        "        for n in Number:\n",
        "            if n.val == next_target_val:\n",
        "                target_number = n\n",
        "                break\n",
        "        \n",
        "        if target_number:\n",
        "            target_card_str = str(suit) + str(target_number)\n",
        "            return target_card_str in hand_card_strs\n",
        "            \n",
        "        return False\n",
        "\n",
        "    def _create_determinized_state(self, original_state, unknown_cards):\n",
        "        \"\"\"フォールバック用のランダム確定化\"\"\"\n",
        "        shuffled_unknown = list(unknown_cards)\n",
        "        random.shuffle(shuffled_unknown)\n",
        "        new_state = original_state.clone()\n",
        "        card_idx = 0\n",
        "        for p_idx in range(new_state.players_num):\n",
        "            if p_idx != self.my_player_num:\n",
        "                count = len(new_state.players_cards[p_idx])\n",
        "                cards_for_p = shuffled_unknown[card_idx : card_idx + count]\n",
        "                new_state.players_cards[p_idx] = Hand(cards_for_p)\n",
        "                card_idx += count\n",
        "        return new_state\n",
        "\n",
        "    def _playout(self, state):\n",
        "        \"\"\"プレイアウト: ゲーム終了まで高速に進める\"\"\"\n",

        "        for a in ais:\n",
        "            a._in_simulation = True\n",
        "\n",
        "        for _ in range(200):\n",
        "            if state.is_done():\n",
        "                break\n",
        "\n",
        "            p_idx = state.turn_player\n",
        "            actions = state.my_actions()\n",

        "        # 勝者を返す\n",
        "        for i, hand in enumerate(state.players_cards):\n",
        "            if len(hand) == 0 and i not in state.out_player:\n",
        "                return i\n",
        "        return -1"

        "\n",
        "### ゲームルール\n",
        "- **プレイヤー数**: 3人対戦\n",
        "- **カード**: 4スーツ × 13枚 = 52枚\n",

        "\n",
        "### AIアルゴリズム\n",
        "**PIMC法 (Perfect Information Monte Carlo)**\n",
